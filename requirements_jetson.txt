# ── Version-sensitive packages ────────────────────────────────────────
# These have tight coupling to torch / CUDA / each other.

numpy>=1.26,<2.0              # torch 2.8 and ultralytics both require numpy <2.0; >=1.26 for ABI compat
opencv-contrib-python==4.10.* # cv2.aruco requires contrib build; plain opencv-python conflicts
ultralytics-thop>=2.0         # ultralytics FLOPs counter (safe to pip-install normally)

matplotlib>=3.5               # optional (GA plotting only); needs numpy <2.0

# ── Packages with no tight version coupling ───────────────────────────
# These work across a wide range of versions. Pinned to installed for
# reproducibility, but loosening is safe.

PyYAML>=5.4               # YAML config parsing
